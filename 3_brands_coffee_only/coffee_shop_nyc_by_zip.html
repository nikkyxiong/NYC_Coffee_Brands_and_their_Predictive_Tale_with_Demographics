library(readr)
library(tidyverse)
library(stringr)

zip_boro <- read_csv('nyc_zip_demo.csv') 

sb <- read_csv('full_nyc_starbucks.csv') %>% 
  select(name = NAME, 
         review_count = ave_review_count, 
         rating = ave_rating, 
         Zipcode = zip_code) %>% 
  left_join(zip_boro) %>% 
  drop_na(Boro)

dd <- read_csv('full_nyc_Dunkin_backup.csv') %>% 
  select(name = NAME, 
         review_count = ave_review_count, 
         rating = ave_rating, 
         Zipcode = zip_code) %>% 
  left_join(zip_boro)%>% 
  drop_na(Boro)
dd$name <- "Dunkin'"

bb <- read_csv('full_nyc_BB.csv') %>% 
  select(name = NAME, 
         review_count = ave_review_count, 
         rating = ave_rating, 
         Zipcode = zip_code) %>% 
  left_join(zip_boro)%>% 
  drop_na(Boro)
bb$name <- "Blue Bottle"

coffee = rbind(sb, dd, bb)
coffee %>% group_by(name) %>% 
    summarize(num = n()) %>% 
    arrange(desc(num))
## # A tibble: 3 x 2
##   name          num

Write dataset by Zipcode

coffee_zip <- coffee %>% 
    group_by(name, Zipcode) %>% 
    summarize(ave_rating = mean(rating, na.rm = T),
           ave_review_count = mean(review_count, na.rm = T),
           num = n()) %>% 
    arrange(desc(num)) %>% 
    select(Name = name,
           Avg_rating = ave_rating,
           Avg_review_count = ave_review_count,
           Number = num,
           Zipcode = Zipcode)

write_csv(coffee_zip, "./coffee_zip.csv", na = "NA" )
