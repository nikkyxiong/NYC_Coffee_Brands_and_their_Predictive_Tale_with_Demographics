library(readr)
library(tidyverse)
library(stringr)

coffee <- read_csv("raw_data/full_nyc_coffee.csv") %>% 
    select(name, review_count, rating, zip_code, 
           borough, city)
# coffee %>% group_by(name) %>% 
#    summarize(num = n()) %>% 
#    arrange(desc(num))
coffee_all <- coffee %>%
  mutate(NAME = case_when(
      str_detect(name, "^Dunkin") ~ "DD",
      str_detect(name, "^Starbucks") ~ "Starbucks",
      str_detect(name, "^Blue Bottle") ~ "Blue Bottle",
      TRUE ~ name
      )) %>% 
    filter(NAME %in% c("DD",
                       "Starbucks", 
                       "Blue Bottle"))

analysis across coffee brands

coffee_all %>% 
    group_by(NAME) %>% 
    summarize(ave_rating = mean(rating, na.rm = T),
           ave_review_count = mean(review_count, na.rm = T),
           num = n()) %>% 
    arrange(desc(num))

dv <- coffee_all %>% 
    group_by(NAME, borough) %>% 
    summarize(ave_rating = mean(rating, na.rm = T),
           ave_review_count = mean(review_count, na.rm = T),
           num = n()) %>% 
    arrange(desc(num))
dv <- dv[-c(11), ]

Write dataset by Zipcode

coffee_all %>% 
    group_by(NAME, zip_code) %>% 
    summarize(ave_rating = mean(rating, na.rm = T),
           ave_review_count = mean(review_count, na.rm = T),
           num = n()) %>% 
    arrange(desc(num)) %>% 
    select(Name = NAME,
           Avg_rating = ave_rating,
           Avg_review_count = ave_review_count,
